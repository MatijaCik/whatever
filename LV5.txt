#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main() {

	int n, i, time;
	printf("Unesite broj procesa:\n");
	scanf("%d", &n);
	int* arrival_times = (int*)calloc(n, sizeof(int));
	int* burst_times = (int*)calloc(n, sizeof(int));
	int* waiting = (int*)calloc(n, sizeof(int));
	int* turnaround = (int*)calloc(n, sizeof(int));
	int* completion = (int*)calloc(n, sizeof(int));
	int* flag = (int*)calloc(n, sizeof(int));

	for (i = 0; i < n; i++)
		scanf("%d", &arrival_times[i]);
	for (i = 0; i < n; i++)
		scanf("%d", &burst_times[i]);
	
	int temp = 0;
	int c = 0;

	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			if (arrival_times[j] < arrival_times[temp]) {
				if(flag[j] != 1){
					temp = j;
				}
			}
			if (j == n - 1 && flag[temp]!= 1) {
				c = c + j + 1;
				completion[temp] = c;
				flag[temp] = 1;
				int k = 0;
				for (k = 0; k < burst_times[temp]; k++) {
					printf("\nIzvrsavanje procesa %d , burst time: %d , flag: %d", temp + 1, k + 1, flag[temp]);
				}
			}
		}
		for (int ha = 0; ha < n; ha++) {
			if (flag[ha] != 1) {
				temp = ha;
			}
		}
	}
	
	return 0;
}
