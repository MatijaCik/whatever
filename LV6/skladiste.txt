CREATE DATABASE IF NOT EXISTS skladiste;
USE skladiste;

CREATE TABLE korisnici (
    id INT AUTO_INCREMENT PRIMARY KEY,
    ime VARCHAR(50),
    prezime VARCHAR(50),
    email VARCHAR(100),
    k_ime VARCHAR(50),
    lozinka VARCHAR(100),
    uloga VARCHAR(20)
);

CREATE TABLE proizvodi (
